def valid?
  temp = 0.chr * 4
  begin
    result = @fmod.invoke('Channel_IsPlaying', @handle, temp)
  rescue
    if (result == FMOD_ERR_INVALID_HANDLE)
      return false
    else
      raise
    end
  end
  # If we get here then it's valid
  return true
end
