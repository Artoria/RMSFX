def default(name)
        case name
        when "$exe"
                Seiran20.api("Kernel32", "GetModuleFileNameW").call 0, buf="\00""*1024, 1024
                Seiran20.to_mb(buf).sub(/\00++$/){}
        when "$dir"
                Seiran20.api("Kernel32", "GetModuleFileNameW").call 0, buf="\00""*1024, 1024
                Seiran20.api("shlwapi", "PathRemoveFileSpecW").call buf
                Seiran20.to_mb(buf).sub(/\00++$/){}
        else
                nil
        end
end

def ini_reader(ini)
    @value = {}
    x = ini.split(/[\r\n]+/).each{|x|
           if /\[(.+)\]/ =~ x
                  @inikey = $1
                  @value[@inikey] ||= {}
           elsif /(.+)=(.+)/=~x
                  @value[@inikey][$1] = $2
           end
    }
    @value
end

def rtp(path)
        path = path.tr("/", "\\")
        @@rtppath.each{|x|
                if FileTest.exist?(name = x + "\\" + path)
                        return name
                end
                Dir.glob(name.tr("\\", "/") + ".*").each{|y|
                        return y.tr("/", "\\")
                }
        }
        nil
end

def which(path)
        x,y = path.split("://")
        if y == nil
                default(x)
        else
                send x, y
        end
end
