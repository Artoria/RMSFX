def load_lit_module(name)
    load_run name
  rescue Object => ex
    print "语言插件载入出错\n\t"+
          "异常 #{ex.to_s}\n\t" +
          "模块 #{name}\n\t"    +
          "回溯\n\t#{ex.backtrace.join("\n")}"
    exit
end

def load_run(name)
  data = File.read(name)
  if data.unpack("C").first>=0x7f
    eval data[3, data.length-3],TOPLEVEL_BINDING,name,1
  else
    eval data,TOPLEVEL_BINDING,name,1
  end
end
