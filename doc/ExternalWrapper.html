 def self.find(dlls)
   dlls.each{|dll|
     (ENV["PATH"].split(";") + ["."]).reverse.each{|x|
       name = x.strip + "/" + dll
       if FileTest.file?(name)
           self.const_set :DLL, name
           return name
       end
     }
  }
